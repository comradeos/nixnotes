Чтобы настроить Vim при запуске, добавьте эти строки в файл ~/.vimrc

set number          " Включить нумерацию строк
syntax on           " Включить подсветку синтаксиса
set tabstop=2       " Количество пробелов при нажатии Tab
set shiftwidth=2    " Шаг автоотступов
set expandtab       " Заменять табуляцию пробелами
